<?php
// Database connection
$servername = "localhost";
$username = "root";
$password = "";
$dbname = "your_database";

$conn = new mysqli($servername, $username, $password, $dbname);

if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
}

// Retrieve form data
$mobile = $_POST['mob'];
$question = $_POST['question'];
$answer = $_POST['answer'];

// Query to verify the user's mobile number and security question answer
$sql = "SELECT * FROM users WHERE mobile = '$mobile' AND $question = '$answer'";
$result = $conn->query($sql);

if ($result->num_rows > 0) {
    echo "Login successful!";
    // Redirect to the user's dashboard or homepage
} else {
    echo "Invalid mobile number or answer to the security question.";
}

$conn->close();
